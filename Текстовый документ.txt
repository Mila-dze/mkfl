#include <iostream>
using namespace std;

int main() {
    int number;
    int sum = 0;
    int count = 0;
    int position = 1;

    cout << "Введите последовательность чисел (введите 0 для завершения):" << endl;

    cin >> number;
    while (number) {
        if (position % 3 == 0&&number % 2 != 0) {
                sum += number;
                count++;
        }

        position++;
        cin >> number;
    }


    if (count > 0) {

        float average = (float)sum / count;
        cout << "Среднее значение нечетных элементов на позициях кратных трем: " << average << endl;
    }
    else {
        cout << "Нет нечетных элементов на позициях кратных трем." << endl;
    }

    return 0;
}
